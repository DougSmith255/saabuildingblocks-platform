'use client';

/**
 * Example usage of TeamHighlights component with WordPress API integration
 *
 * This file demonstrates how to:
 * 1. Fetch highlights data from WordPress REST API
 * 2. Handle loading and error states
 * 3. Use the TeamHighlights component with fetched data
 */

import React, { useEffect, useState } from 'react';
import { TeamHighlights } from '@/components/homepage/TeamHighlights';
import type { Highlight } from '@/types/homepage';

interface HomepageApiResponse {
  highlights: Highlight[];
}

export default function TeamHighlightsExample() {
  const [highlights, setHighlights] = useState<Highlight[]>([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    async function fetchHighlights() {
      try {
        setLoading(true);
        setError(null);

        // Replace with your WordPress site URL
        const response = await fetch(
          `${process.env.NEXT_PUBLIC_WP_API_URL}/wp-json/saa/v1/homepage`
        );

        if (!response.ok) {
          throw new Error(`HTTP error! status: ${response.status}`);
        }

        const data: HomepageApiResponse = await response.json();
        setHighlights(data.highlights);
      } catch (err) {
        console.error('Error fetching highlights:', err);
        setError(err instanceof Error ? err.message : 'Failed to fetch highlights');
      } finally {
        setLoading(false);
      }
    }

    fetchHighlights();
  }, []);

  if (loading) {
    return (
      <div className="min-h-screen flex items-center justify-center">
        <div className="text-center">
          <div className="animate-spin rounded-full h-12 w-12 border-b-2 border-[#D4AF37] mx-auto mb-4"></div>
          <p className="text-muted-foreground">Loading highlights...</p>
        </div>
      </div>
    );
  }

  if (error) {
    return (
      <div className="min-h-screen flex items-center justify-center">
        <div className="text-center text-red-500">
          <p className="text-xl font-semibold mb-2">Error Loading Highlights</p>
          <p className="text-sm">{error}</p>
        </div>
      </div>
    );
  }

  return (
    <main className="min-h-screen">
      <TeamHighlights highlights={highlights} />
    </main>
  );
}
